组合继承的实例

        
function Person (name, age) {

    this.name = name;
    this.age = age;

}

Person.prototype.eat = function () {
    console.log("i like eat");
}

/*
    这里用 apply 来使用 Person 的构造(适用于属性)
    和下面一起组合成 组合继承（下面的适用于方法）
*/
function Student (name, age) {
    Person.apply(this, [name, age]);
}

//  将子对象实例的 prototype 属性指向父对象的实例
Student.prototype = new Person();
//  将 constructor 属性指向本来的对象
Student.prototype.constructor = Student;

var stu1 = new Student("ch", 20);
console.log(stu1.eat);
console.log(stu1);
